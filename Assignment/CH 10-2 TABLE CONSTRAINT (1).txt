1. CREATE TABLE TO STORE BELOW INFO.
   FACULTIES : FACULTY_NO, FACNM, JOINDATE, SALARY
   
   CREATE TABLE FACULTIES(FACULTY_NO INT PRIMARY KEY,
   FACNM VARCHAR(30) NOT NULL,
   JOINDATE DATE NOT NULL,
   SALARY INT CHECK(SALARY BETWEEN 20000 AND 80000));
	
2. CREATE TABLE TO STORE BELOW INFO.
   SUBJECTS : SUBJECT_NO, SUBNM, TOTAL_DURATION_IN_HOURS
   
   CREATE TABLE SUBJECTS(SUBJECT_NO INT UNIQUE,
   SUBNM VARCHAR(30) NOT NULL,
   TOTAL_DURATION_IN_HOURS INTERVAL DAY TO SECOND NOT NULL);

3. CREATE TABLE TO STORE BELOW INFO.
   FACULTY_SUBJECT : FACULTY_NO, SUBJECT_NO

   CREATE TABLE FACULTY_SUBEJCT(FACULTY_NO INT CONSTRAINT SCOTT_FACULTY_SUB_FACULTY_NO_PK PRIMARY KEY,
   SUBJECT_NO INT UNIQUE);

4. CREATE TABLE TO STORE BELOW INFO.
   COURSE : COURSE_NO, COURSE_NAME, DURATION_IN_MONTHS, GENERAL_CLUB, TOTAL_FEES
   
   CREATE TABLE COURSE(COURSE_NO INT PRIMARY KEY,
   COURSE_NAME VARCHAR(30) NOT NULL,
   DURATION_IN_MONTHS INTERVAL YEAR TO MONTH NOT NULL,
   TOTAL_FEES INT,
   GENERAL_CLUB VARCHAR(10) NOT NULL CONSTRAINT SCOTT_COURSE_GENERAL_CLUB_CK#1 CHECK(GENERAL_CLUB IN ('GENERAL','CLUB')));

5. CREATE TABLE TO STORE BELOW INFO.
   STUDENTS : STUDENT_NO, STUNM, BIRTH_DATE, COURSE_NO

   CREATE TABLE STUDENTS(STUDENT_NO INT PRIMARY KEY,
   STUNM VARCHAR(30) NOT NULL,
   BIRTH_DATE DATE NOT NULL,
   COURSE_NO INT NOT NULL);

6. CREATE TABLE TO STORE BELOW INFO.
   FEES : RECEIPT_NO, RECDT, AMOUNT, CASH_OR_CHEQUE, CHEQUE_NO, CHEQUE_DATE, BANK_NAME

   CREATE TABLE FEES(RECEIPT_NO INT UNIQUE,
   BANK_NAME VARCHAR(30) NOT NULL,
   RECDT DATE NOT NULL,
   AMOUNT INT NOT NULL,
   CHEQUE_DATE DATE,
   CHEQUE_NO INT UNIQUE,
   CASH_OR_CHEQUE VARCHAR(10));

7. CREATE TABLE TO STORE BELOW INFO.
   MARKS : STUDENT_NO, SUBJECT_NO, FACULTY_NO, MARKS_SCORED

   CREATE TABLE MARKS(STUDENT_NO INT PRIMARY KEY,
   SUBJECT_NO INT UNIQUE,
   FACULTY_NO INT UNIQUE,
   MARKS_SCORED INT NOT NULL);

8. CREATE TABLE TO STORE BELOW INFO.
   BATCH_MASTER : BATCH_NO, FACULTY_NO, SUBJECT_NO, START_DT, DURATION_IN_DAYS

   CREATE TABLE BATCH_MASTER(BATCH_NO INT PRIMARY KEY,
   DURATION_IN_DAYS INTERVAL DAY TO SECOND NOT NULL,
   START_DT DATE NOT NULL,
   SUBJECT_NO INT CONSTRAINT SCOTT_BATCH_MASTER_SUBJECT_NO_REF_MARKS_SUBJECT_NO REFERENCES MARKS(SUBJECT_NO),
   FACULTY_NO INT);
 
   SELECT TABLE_NAME,CONSTRAINT_TYPE,CONSTRAINT_NAME FROM USER_CONSTRAINTS WHERE TABLE_NAME = 'BATCH_MASTER';         // OP BOLTE!!!!!!!!!!
