1 CREATE IDENTICAL TABLE OF REGIONS WITH DEFINITION AND DATA AND NAME IT R1
CREATE TABLE R1 AS SELECT * FROM REGIONS;

2 CREATE TABLE E2 USING EMPLOYEES TABLE WHICH WILL HAVE EMPLOYEE_ID , JOB_ID , SALARY COLUMNS ONLY.
CREATE TABLE E2 AS SELECT EMPLOYEE_ID,JOB_ID,SALARY FROM EMPLOYEES;

3 CREATE TABLE E3 USING EMPLOYEES TABLE WHICH WILL HAVE ONLY ROWS HAVING SALARY HIGHER THAN 10000.
CREATE TABLE E3 AS SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY>10000;

4 CREATE TABLE E4 USING EMPLOYEES TABLE WHICH WILL HAVE FIRST_NAME , SALARY , JOB_ID , HIRE_DATE COLUMNS. INCLUDE ONLY THOSE ROWS WHICH WERE HIRED BEFORE 1995.
CREATE TABLE E4 AS SELECT FIRST_NAME,JOB_ID,SALARY,HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE < '01-JAN-1995';

5 CREATE TABLE D5 USING DEPARTMENTS TABLE. ONLY DEFINITION SHOULD BE CREATED WITHOUT ANY DATA.
CREATE TABLE D5 AS SELECT * FROM DEPARTMENTS WHERE DEPARTMENT_ID=-1;        // NO ROWS WILL BE SELECTED

6 CREATE TABLE ED6 USING EMPLOYEES AND DEPARTMENTS TABLES. INCLUDE FIRST_NAME AND DEPARTMENT_NAME WITH ALL ROWS.
CREATE TABLE ED6 AS SELECT FIRST_NAME,DEPARTMENT_NAME FROM EMPLOYEES NATURAL JOIN DEPARTMENTS;       

7 CREATE TABLE E7 USING EMPLOYEES TABLE. INCLUDE JOB_ID AND HIGHEST SALRAY PAID TO EACH JOB_ID.
CREATE TABLE E7 AS SELECT JOB_ID,MAX(SALARY) FROM EMPLOYEES GROUP BY JOB_ID;

8 CREATE TABLE E8 USING EMPLOYEES TABLE. INCLUDE FIRST_NAME AND ANNUAL SALARY. 
CREATE TABLE E8 AS SELECT FIRST_NAME,SALARY*12 "ANNUAL SALARY" FROM EMPLOYEES;

9 CREATE TABLE E9 USING EMPLOYEES AND DEPARTMENTS TABLES. INCLUDE ROWS WHICH WORKS IN Shiping DEPARTMENT.
CREATE TABLE E9 AS SELECT FIRST_NAME,DEPARTMENT_ID FROM EMPLOYEES WHERE DEPARTMENT_ID IN (SELECT DEPARTMENT_ID FROM DEPARTMENTS WHERE DEPARTMENT_NAME='Shiping');

10 CREATE TABLE E10 USING DEPARTMENTS AND EMPLOYEES TABLES. INCLUDE DEPARTMENT_NAME , LOWEST SALARY , HIGHEST SALARY PAID TO EACH DEPARTMENT.
CREATE TABLE E10 AS SELECT D.DEPARTMENT_NAME,MAX(SALARY) MAX_SAL,MIN(SALARY) MIN_SAL FROM EMPLOYEES E 
JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID GROUP BY D.DEPARTMENT_NAME;


11 CREATE TABLE E11 WHICH WILL HAVE CAPITAL FIRST_NAME , LOWER EMAIL 
CREATE TABLE E11 AS SELECT UPPER(FIRST_NAME),LOWER(EMAIL) FROM EMPLOYEES;

12 CREATE TABLE E12 WHICH WILL HAVE LAST_NAME , DEPARTMENT_NAME
CREATE TABLE E12 AS SELECT LAST_NAME,DEPARTMENT_NAME FROM EMPLOYEES NATURAL JOIN DEPARTMENTS;

13 CREATE TABLE E13 WHICH HAS COLUMN NAMES EDNAME. THIS SHOULD STORE ALL FIRST_NAME AND DEPARTMENT_NAME.
CREATE TABLE E13 AS SELECT FIRST_NAME "EDNAME" FROM EMPLOYEES UNION 
SELECT DEPARTMENT_NAME FROM DEPARTMENTS;

14 CREATE TABLE E14 WHICH WILL HAVE DEPARTMENT_NAME AND HIGHEST SALARY PAID FOR EACH.
CREATE TABLE E14 AS SELECT D.DEPARTMENT_NAME,MAX(SALARY) MAX_SAL FROM EMPLOYEES E
JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID GROUP BY D.DEPARTMENT_NAME;
 
15 CREATE TABLE E15 WHICH WILL STORE ALL COLUMNS OF EMPLOYEES TABLE BUT ONLY ROWS BETWEEN 12 AND 18.
CREATE TABLE E15 AS SELECT * FROM EMPLOYEES OFFSET 11 ROWS FETCH NEXT 7 ROWS ONLY;

16 CREATE TABLE E16 WHICH WILL STORE MAX SALARY FOR EACH GRADE LEVEL  // JODD BOLTE!!

CREATE TABLE E16 AS SELECT GRADE_LEVEL,MAX(SALARY) FROM EMPLOYEES 
JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL GROUP BY GRADE_LEVEL;
