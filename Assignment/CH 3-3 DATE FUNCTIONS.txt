3 COMPANY PAYS SALARY ON LAST DAY OF CURRENT MONTH, DISPLAY FIRST SALARY DATE
SELECT FIRST_NAME , HIRE_DATE , LAST_DAY(HIRE_DATE) FROM EMPLOYEES;
  
4 COMPANY PAYS SALARY ON NEXT MONTH'S FIRST DATE, DISPLAY FIRST SALARY DATE
SELECT FIRST_NAME , HIRE_DATE , LAST_DAY(HIRE_DATE)+1 FROM EMPLOYEES;
  
5 DISPLAY HOW MANY DAYS EACH EMPLOYEE HAS WORKED IN THE MONTH OF HIS JONING
SELECT FIRST_NAME, HIRE_DATE, TRUNC(SYSDATE-HIRE_DATE) FROM EMPLOYEES;
  
6 DISPLAY FOR HOW MANY COMPLETED YEARS EACH EMPLOYEE IS WORKING WITH COMPANY
SELECT FIRST_NAME, HIRE_DATE, TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12) COMPLETED_YEARS FROM EMPLOYEES; 
 
7 COMPANY PAYS FIRST INCREMENT AFTER 6 MONTHS OF JOINING, DISPLAY EACH EMPLOYEE'S FIRST INCREMENT DATE
SELECT FIRST_NAME, HIRE_DATE, ADD_MONTHS(HIRE_DATE,6) FROM EMPLOYEES;

15 COMPANY PAYS INCENTIVES ON LAST DATE OF EACH CALENDER QUARTER. FIND FIRST INCENTIVE DATE.
SELECT FIRST_NAME,LAST_DAY(ADD_MONTHS(TRUNC(SYSDATE,'Q')),3) FROM EMPLOYEES;

17 COMPANY PAYS SALARY ON FIRST WEDNESDAY OF EACH MONTH. FIND FIRST AND LAST SALARY DATE. DISPLAY DAY NAME WITH DATES.

18 COMPANY PAYS ON LAST SUNDAY OF EACH MONTH. FIND FIRST AND NEXT SALARY DATE. DISPLAY DAY NAME WITH DATES.

19 COMPANY PAYS BONUS TWO TIMES PER YEAR ON LAST DATE OF 2nd AND 4th QUARTER OF CALENDER YEAR. FIND FIRST BONUS DATE.  // JODD BOLTE RE!!
SELECT FIRST_NAME,HIRE_DATE,TO_CHAR(HIRE_DATE,'RR'),
CASE WHEN TO_CHAR(HIRE_DATE,'Q') IN (2,4) THEN ADD_MONTHS(TRUNC(HIRE_DATE,'Q'),3)-1
     WHEN TO_CHAR(HIRE_DATE,'Q') IN (1,3) THEN ADD_MONTHS(TRUNC(HIRE_DATE,'Q'),6)-1 END "FIRST BONUS DATE" FROM EMPLOYEES;


20 COMPANY DECLARES INCREMENTS TWO TIMES PER YEAR ON FIRST DATE OF 1st AND 3rd QUARTER AS PER ACCOUNTING YEAR. FIND SECOND INCREMENT DATE FOR EACH EMPLOYEE.        OP QUESTION BOLTE RE!!!!
SELECT FIRST_NAME,HIRE_DATE,TO_CHAR(HIRE_DATE,'RR'),
CASE WHEN TO_CHAR(HIRE_DATE,'Q') IN (2,4) THEN ADD_MONTHS(TRUNC(HIRE_DATE,'Q'),9)
     WHEN TO_CHAR(HIRE_DATE,'Q') IN (1,3) THEN ADD_MONTHS(TRUNC(HIRE_DATE,'Q'),12) END "SECOND INCREMENT DATE" FROM EMPLOYEES;
