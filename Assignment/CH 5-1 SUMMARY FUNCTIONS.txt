1COUNT TOTAL NO. OF EMPLOYEES WORK IN COMPANY

2DISPLAY HIGHEST SALARY PAID FIGURE
SELECT MAX(SALARY) FROM EMPLOYEES;

3DISPLAY THE DATE ON WHICH FIRST EMPLOYEE WAS HIRED
SELECT MIN(HIRE_DATE) FROM EMPLOYEES;

4DISPLAY AVERAGE SALARY PAID TO DEPARTMENT 60
SELECT AVG(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=60;

5DISPLAY LOWEST SALARY PAID TO IT_PROG
SELECT MIN(SALARY) FROM EMPLOYEES WHERE JOB_ID='IT_PROG';

6DISPLAY COUNT OF EMPLOYEES WHO REPORT TO MANAGER_ID 100
SELECT COUNT(*) FROM EMPLOYEES WHERE MANAGER_ID=100;

7DISPLAY COUNT OF EMPLOYEES WHO WERE HIRED BETWEEN YEAR 91 AND 95
SELECT COUNT(*) FROM EMPLOYEES WHERE HIRE_DATE BETWEEN '01-JAN-91' AND '31-DEC-95';

8DISPLAY COUNT OF EMPLOYEES WHO GETS SALARY UPTO 6000
SELECT COUNT(*) FROM EMPLOYEES WHERE SALARY<=6000;

10COUNT UNIQUE MANAGER_ID
