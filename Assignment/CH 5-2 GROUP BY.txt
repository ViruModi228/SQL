2 DISPLAY JOB_WISE LOWEST AND HIGHEST SALARY

3 DISPLAY MANAGER_ID WISE AVERAGE SALARY
SELECT MANAGER_ID ,AVG(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID;

4 DISPLAY DEPARTMENT_ID AND JOB_ID WISE NO. OF EMPLOYEES
SELECT DEPARTMENT_ID,JOB_ID, COUNT(*) FROM EMPLOYEES GROUP BY DEPARTMENT_ID,JOB_ID;

5 DISPLAY HIRE YEAR WISE NO. OF EMPLOYEES
SELECT TO_CHAR(HIRE_DATE,'RR'),COUNT(*) FROM EMPLOYEES GROUP BY(TO_CHAR(HIRE_DATE,'RR'));

6 DISPLAY QUARTER WISE TOTAL OF SALARY
SELECT TO_CHAR(HIRE_DATE,'Q'),SUM(SALARY) FROM EMPLOYEES GROUP BY (TO_CHAR(HIRE_DATE,'Q'));

7 DISPLAY DEPARTMENT_ID WISE, JOB_ID WISE LOWEST AND HIGHEST SALARY // ALL THE NON SUMMARY ITEMS PROJECTED MUST BE A PART OF GROUP BY CLAUSE!!!!!!
SELECT DEPARTMENT_ID,JOB_ID ,MAX(SALARY),MIN(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID,JOB_ID;

8 DISPLAY MANAGER_ID WISE , JOB_ID WISE AVERAGE SALARY AND TOTAL OF SALARY
SELECT MANAGER_ID,JOB_ID,AVG(SALARY),SUM(SALARY) FROM EMPLOYEES GROUP BY MANGER_ID,JOB_ID;

9 DISPLAY DEPARTMENT_ID WISE , MANAGER_ID WISE COUNT OF EMPLOYEES
SELECT MANAGER_ID,DEPARTMENT_ID,COUNT(*) FROM EMPLOYEES GROUP BY MANGER_ID,DEPARTMENT_ID;

10 DISPLAY DEPARTMENT_ID WISE , JOB_ID WISE , MANAGER_ID WISE LOWEST SALARY, AVERAGE SALARY AND HIGHEST SALARY
SELECT DEPARTMENT_ID,JOB_ID,MANAGER_ID,AVG(SALARY),MAX(SALARY),MIN(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID,JOB_ID,DEPARTMENT_ID;

11 DISPLAY ONLY DATE WISE (IGNORE MONTH AND YEAR) HIGHEST AND LOWEST SALARY
SELECT TO_CHAR(HIRE_DATE,'DD'),MAX(SALARY),MIN(SALARY) FROM EMPLOYEES GROUP_BY(TO_CHAR(HIRE_DATE,'DD'));  //JODD BEE!!!

12 DISPLAY DAY WISE (MON,TUE...) AVERAGE SALARY
SELECT TO_CHAR(HIRE_DATE,'DAY'),AVG(SALARY) FROM EMPLOYEES GROUP BY(TO_CHAR(HIRE_DATE,'DAY'));

13 DISPLAY MONTH WISE (IGNORE DATE AND YEAR) TOTAL SALARY
SELECT TO_CHAR(HIRE_DATE,'MON'),SUM(SALARY) FROM EMPLOYEES GROUP BY(TO_CHAR(HIRE_DATE,'MON'));
