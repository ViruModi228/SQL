1 DISPLAY COUNT OF EMPLOYEES FOR DEPARTMENT_ID DEPT90 , DEPT50 DEPT60 (CROSS TAB REPORT)
SELECT COUNT(DECODE(DEPARTMENT_ID,90,DEPARTMENT_ID)) DEPT_50,
       COUNT(DECODE(DEPARTMENT_ID,90,DEPARTMENT_ID)) DEPT_90,
       COUNT(DECODE(DEPARTMENT_ID,90,DEPARTMENT_ID)) DEPT_60 FROM EMPLOYEES;

2 DISPLAY LOWEST SALARY OF EMPLOYEES FOR JOB_ID IT_PROG , ST_CLERK , SA_REP , AD_VP (PIVOTE REPORT)
SELECT MIN(DECODE(JOB_ID,'IT_PROG',SALARY)) MIN_IT_PROG,
       MIN(DECODE(JOB_ID,'ST_CLERK',SALARY)) MIN_ST_CLERK,
       MIN(DECODE(JOB_ID,'SA_REP',SALARY)) MIN_SA_REP,
       MIN(DECODE(JOB_ID,'AD_VP',SALARY)) MIN_AD_VP FROM EMPLOYEES;

3 DISPLAY HIGHEST SALARY OF EMPLOYEES FOR MANAGER_ID 100 , 124 , 149 (CROSS TAB REPORT)
SELECT MAX(DECODE(MANAGER_ID,100,SALARY)) MAX_100,
       MAX(DECODE(MANAGER_ID,124,SALARY)) MAX_124,
       MAX(DECODE(MANAGER_ID,149,SALARY)) MAX_149
       FROM EMPLOYEES;

4 DISPLAY DEPARTMENT_ID WISE AVERAGE SALARY FOR IT_PROG , SA_REP , AD_VP , ST_CLERK (MATRIX REPORT)
SELECT DEPARTMENT_ID,AVG(DECODE(JOB_ID,'IT_PROG',SALARY)) AVG_IT_PROG,
                     AVG(DECODE(JOB_ID,'ST_CLERK',SALARY)) AVG_ST_CLERK,
       		     AVG(DECODE(JOB_ID,'SA_REP',SALARY)) AVG_SA_REP,
       		     AVG(DECODE(JOB_ID,'AD_VP',SALARY)) AVG_AD_VP FROM EMPLOYEES GROUP BY DEPARTMENT_ID;

SELECT DEPARTMENT_ID,AVG(DECODE(JOB_ID,'IT-PROG',SALARY)) IT_PROG_AVG_SAL,
                     AVG(CASE JOB_ID WHEN 'AD_VP' THEN SALARY END) AD_VP_AVG_SAL,
                     AVG(CASE WHEN JOB_ID = 'SA_MAN' THEN SALARY END) SA_MAN_AVG_SAL,
                     AVG(DECODE(JOB_ID,'ST_CLERK',SALARY)) ST_CLERK_AVG_SAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID;

TO CHECK OR VERIFY REPORT GENERATED:-
                     
SELECT DEPARTMENT_ID,JOB_ID,AVG(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID,JOB_ID HAVING DEPARTMENT_ID = 90 AND JOB_ID = 'AD_VP'; 

5 DISPLAY MANAGER_ID WISE TOTAL OF SALARY FOR DEPARTMENT_ID 50,60,80,90 (PIVOTE REPORT)
SELECT MANAGER_ID,SUM(DECODE(DEPARTMENT_ID,50,SALARY)) SUM_50,
                  SUM(DECODE(DEPARTMENT_ID,60,SALARY)) SUM_60,
                  SUM(DECODE(DEPARTMENT_ID,80,SALARY)) SUM_80,
                  SUM(DECODE(DEPARTMENT_ID,90,SALARY)) SUM_90
                  FROM EMPLOYEES GROUP BY MANAGER_ID;

6 DISPLAY DEPARTMENT_ID WISE TOTAL OF SALARY FOR IT_PROG , SA_RAP , AD_VP , ST_CLERK ONLY FOR DEPARTMENT_ID 90,50,80,60 (MATRIX REPORT)
SELECT DEPARTMENT_ID,SUM(DECODE(JOB_ID,'IT_PROG',SALARY)) SUM_IT_PROG,
                     SUM(DECODE(JOB_ID,'ST_CLERK',SALARY)) SUM_ST_CLERK,
       		     SUM(DECODE(JOB_ID,'SA_REP',SALARY)) SUM_SA_REP,
       		     SUM(DECODE(JOB_ID,'AD_VP',SALARY)) SUM_AD_VP FROM EMPLOYEES GROUP BY DEPARTMENT_ID HAVING DEPARTMENT_ID =SOME(90,50,80,60);
TO-VERIFY ANSWER:-

SELECT DEPARTMENT_ID,JOB_ID,SUM(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID,JOB_ID HAVING DEPARTMENT_ID = 90 AND JOB_ID = 'AD_VP';
