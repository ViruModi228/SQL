1 DISPLAY CITY , COUNTRY_NAME (USING)
SELECT CITY,COUNTRY_NAME FROM LOCATIONS JOIN COUNTRIES USING(COUNTRY_ID);

2 DISPLAY COUNTRY_NAME , REGION_NAME (ON)
SELECT COUNTRY_NAME,REGION_NAME FROM COUNTRIES JOIN REGIONS USING(REGION_ID);

3 DISPLAY DEPARTMENT_NAME , CITY , COUNTRY_NAME (USING)
SELECT DEPARTMENT_NAME,CITY,COUNTRY_NAME FROM DEPARTMENTS
JOIN LOCATIONS USING(LOCATION_ID) 
JOIN COUNTRIES USING(COUNTRY_ID);

4 DISPLAY DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME (ON)
SELECT D.DEPARTMENT_NAME,L.CITY,C.COUNTRY_NAME,R.REGION_NAME FROM DEPARTMENTS D
JOIN LOCATIONS L ON D.LOCATION_ID=L.LOCATION_ID 
JOIN COUNTRIES C ON L.COUNTRY_ID=C.COUNTRY_ID;

6 DISPLAY DEPARTMENT_NAME , REGION_NAME (ON)

11 DISPLAY FIRST_NAME , JOB_TITLE , SALARY , GRADE_LEVEL , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME
SELECT FIRST_NAME,JOB_TITLE,SALARY,GRADE_LEVEL,DEPARTMENT_NAME,CITY,COUNTRY_NAME,REGION_NAME FROM EMPLOYEES E
JOIN DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
JOIN LOCATIONS L ON D.LOCATION_ID=L.LOCATION_ID
JOIN COUNTRIES C ON L.COUNTRY_ID=C.COUNTRY_ID
JOIN REGIONS R ON C.REGION_ID=R.REGION_ID
JOIN JOBS J ON E.JOB_ID=J.JOB_ID
JOIN JOB_GRADES ON SALARY BETWEEN LOWEST_SAL AND HIGHEST_SAL;
