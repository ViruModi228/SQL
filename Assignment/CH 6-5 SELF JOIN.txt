1 DISPLAY EMPLOYEE'S FIRST_NAME AND MANAGER'S FIRST_NAME            // SAARE TABLES KA JOIN WE HAVE DID IT IN THE LAST SO NOW IT IS VERY SIMPULLL FOR US!!!!!!!!!!!!!!!
SELECT E.FIRST_NAME EMP_NAME,M.FIRST_NAME MGR_NAME FROM EMPLOYEES E
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID;

2 DISPLAY EMPLOYEE'S FIRST_NAME AND LAST_NAME  WITH MANAGER'S FIRST_NAME AND LAST_NAME
SELECT E.FIRST_NAME EMP_NAME,M.FIRST_NAME MGR_NAME,E.LAST_NAME EMP_LNAME,M.LAST_NAME MGR_LNAME FROM EMPLOYEES E
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID;

3 DISPLAY EMPLOYEE'S FIRST_NAME , SALARY WITH MANAGER'S FIRST_NAME AND SALARY
SELECT E.FIRST_NAME EMP_NAME,M.FIRST_NAME MGR_NAME,E.SALARY EMP_SAL,M.SALARY MGR_SAL FROM EMPLOYEES E
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID;

4 DISPLAY EMPLOYEE'S LAST_NAME AND HIRE_DATE WITH MANAGER'S FIRST_NAME AND HIRE_DATE
SELECT E.LAST_NAME EMP_LNAME,M.FIRST_NAME MGR_NAME,E.HIRE_DATE EMP_HIRE_DATE,M.HIRE_DATE MGR_HIRE_DATE FROM EMPLOYEES E
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID;

5 DISPLAY EMPLOYEE'S FIRST_NAME AND JOB_TITLE WITH MANAGER' FIRST_NAME AND JOB_TITLE
SELECT E.FIRST_NAME EMP_NAME,EJ.JOB_TIT EMP_JOB_TITLE,M.FIRST_NAME MGR_NAME,MJ.JOB_TITLE MGR_JOB_TITLE FROM EMPLOYEES E
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN JOBS EJ ON E.JOB_ID=EJ.JOB_ID
JOIN JOBS MJ ON M.JOB_ID=MJ.JOB_ID;

6 DISPLAY EMPLOYEE'S LAST_NAME , SALARY , GRADE_LEVEL WITH MANAGER'S FIRST_NAME , SALARY , GRADE_LEVEL
SELECT E.LAST_NAME EMP_LNAME,E.SALARY EMP_SAL,EJG.GRADE_LEVEL EMP_GRADE_LEVEL,
M.FIRST_NAME EMP_NAME,M.SALARY MGR_SAL,MJG.GRADE_LEVEL MGR_GRADE_LEVEL
FROM EMPLOYEES E JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN JOB_GRADES EJG ON E.SALARY BETWEEN EJG.LOWEST_SAL AND EJG.HIGHEST_SAL
JOIN JOB_GRADES MJG ON M.SALARY BETWEEN MJG.LOWEST_SAL AND MJG.HIGHEST_SAL;

7 DISPLAY EMPLOYEE'S NAME AND EMPLOYEE'S DEPARTMENT_NAME  , MANAGER'S NAME AND MANAGER'S DEPARTMENT_NAME
SELECT E.FIRST_NAME||E.LAST_NAME EMP_NAME,M.FIRST_NAME||M.LAST_NAME MGR_NAME,ED.DEPARTMENT_NAME EMP_DNAME,MD.DEPARTMENT_NAME MGR_DNAME
FROM EMPLOYEES E JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN DEPARTMENTS ED ON E.DEPARTMENT_ID=ED.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON M.DEPARTMENT_ID=MD.DEPARTMENT_ID;


8 DISPLAY EMPLOYEE'S NAME , EMPLOYEE'S CITY , MANAGER'S NAME , MANAGER'S CITY
SELECT E.FIRST_NAME||E.LAST_NAME EMP_NAME,M.FIRST_NAME||M.LAST_NAME MGR_NAME,EL.CITY EMP_CITY,ML.CITY MGR_CITY FROM EMPLOYEES E 
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN DEPARTMENTS ED ON E.DEPARTMENT_ID=ED.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON M.DEPARTMENT_ID=MD.DEPARTMENT_ID
JOIN LOCATIONS EL ON ED.LOCATION_ID=EL.LOCATION_ID
JOIN LOCATIONS ML ON MD.LOCATION_ID=ML.LOCATION_ID;

9 DISPLAY EMPLOYEE'S NAME , EMPLOYEE'S REGION_NAME , MANAGER'S NAME , MANAGER'S REGION_NAME
SELECT E.FIRST_NAME||E.LAST_NAME EMP_NAME,M.FIRST_NAME||M.LAST_NAME MGR_NAME,ER.REGION_NAME EMP_RNAME,MR.REGION_NAME MGR_RNAME FROM EMPLOYEES E 
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN DEPARTMENTS ED ON E.DEPARTMENT_ID=ED.DEPARTMENT_ID
JOIN DEPARTMENTS MD ON M.DEPARTMENT_ID=MD.DEPARTMENT_ID
JOIN LOCATIONS EL ON ED.LOCATION_ID=EL.LOCATION_ID
JOIN LOCATIONS ML ON MD.LOCATION_ID=ML.LOCATION_ID
JOIN COUNTRIES EC ON EL.COUNTRY_ID=EC.COUNTRY_ID
JOIN COUNTRIES MC ON ML.COUNTRY_ID=MC.COUNTRY_ID
JOIN REGIONS ER ON EC.REGION_ID=ER.REGION_ID
JOIN REGIONS MR ON MC.REGION_ID=MR.REGION_ID;

10 DISPLAY EMPLOYEE NAME , EMPLOYEE SALARY , EMPLOYEE GRADE_LEVEL , EMPLOYEE JOB_TITLE , MANAGER NAME , MANAGER SALARY , MANAGER GRADE_LEVEL  MANAGER JOB_TITLE
SELECT E.FIRST_NAME||E.LAST_NAME EMP_NAME,M.FIRST_NAME||M.LAST_NAME MGR_NAME,E.SALARY EMP_SAL,M.SALARY MGR_SALARY,
EJG.GRADE_LEVEL EMP_GRADE_LEVEL,MJG.GRADE_LEVEL MGR_GRADE_LEVEL,EJ.JOB_TITLE EMP_JOB_TITLE,MJ.JOB_TITLE MGR_JOB_TITLE FROM EMPLOYEES E 
JOIN EMPLOYEES M ON E.MANAGER_ID=M.EMPLOYEE_ID
JOIN JOB_GRADES EJG ON E.SALARY BETWEEN EJG.LOWEST_SAL AND EJG.HIGHEST_SAL
JOIN JOB_GRADES MJG ON M.SALARY BETWEEN MJG.LOWEST_SAL AND MJG.HIGHEST_SAL   
JOIN JOBS EJ ON E.JOB_ID=EJ.JOB_ID
JOIN JOBS MJ ON M.JOB_ID=MJ.JOB_ID;

EJ.JOB_TITLE EMP_JOB_TITLE,EGJ.GRADE_LEVEL EMP_GRADE_LEVEL,M.FIRST_NAME MGR_FNAME,M.SALARY MGR_SALARY,MD.DEPARTMENT_NAME MGR_DNAME,ML.CITY MGR_CITY,MC.COUNTRY_NAME MGR_COUNTRY,MR.REGION_NAME MGR_RNAME,
