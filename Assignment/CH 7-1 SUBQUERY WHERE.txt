1 DISPLAY THE EMPLOYEE INFO. WHO GETS HIGHER SALARY THAN AVERAGE SALARY PAID
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES);

2 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED FIRST IN COMPANY
SELECT FIRST_NAME,HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE = (SELECT MIN(HIRE_DATE) FROM EMPLOYEES);

3 DISPLAY THE EMPLOYEE INFO. WHO WERE HIRED LAST IN COMPANY
SELECT FIRST_NAME,HIRE_DATE FROM EMPLOYEES WHERE HIRE_DATE = (SELECT MAX(HIRE_DATE) FROM EMPLOYEES);

4 DISPLAY INFO. OF EMPLOYEE WHO GETS HIGHEST COMMISSION AMOUNT
SELECT FIRST_NAME,SALARY*COMMISSION_PCT "COMMISSION AMOUNT" FROM EMPLOYEES WHERE SALARY*COMMISSON_PCT = (SELECT MAX(SALARY*COMMSSION_PCT) FROM EMPLOYEES);

5 DISPLAY INFO. OF EMPLOYEE WHO GETS LOWEST COMMISSION AMOUNT
SELECT FIRST_NAME,SALARY*COMMISSION_PCT "COMMISSION AMOUNT" FROM EMPLOYEES WHERE SALARY*COMMISSON_PCT = (SELECT MIN(SALARY*COMMSSION_PCT) FROM EMPLOYEES);

6 DISPLAY THOSE WHO GETS LOWER SALARY THAN AVERAGE SALARY PAID TO IT_PROG
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY < (SELECT AVG(SALARY) FROM EMPLOYEES WHERE JOB_ID='IT_PROG');

7 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 80
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=80);

8 DISPLAY THOSE WHO GETS LOWER SALARY THAN ALL EMPLOYEES OF DEPARTMENT_ID 60
SELECT FIRST_NAME,SALARY,DEPARTMENT_ID FROM EMPLOYEES WHERE SALARY < (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=60);

9 DISPLAY THOSE WHO GETS HIGHER SALARY THAN ANY SA_REP
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY > ANY(SELECT SALARY FROM EMPLOYEES WHERE JOB_ID='SA_REP');

10 DISPLAY THOSE WHO GETS LOWER SALARY THAN LOWEST SALARY PAID WHO REPORTS TO 149 (USE MULTI ROW COMPARISON OPERATOR)
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY <(SELECT MIN(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=149);

11 DISPLAY THOSE WHO GETS HIGHER SALARY THAN HIGHEST SALARY PAID TO DEPARTMENT_ID 20 (USE MULTI ROW COMPARISON OPERATOR)
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE SALARY > (SELECT MAX(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID=20);

12 DISPLAY ALL OTHER THAN THOSE WHO GETS HIGHEST SALARY IN THEIR DEPARTMENT
SELECT FIRST_NAME,SALARY FROM EMPLOYEES WHERE (DEPARTMENT_ID,SALARY) ^= ALL(SELECT DEPARTMENT_ID,MAX(SALARY) FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

13 DISPLAY ALL OTHER THAN THOSE WHO GETS LOWEST SALARY FOR THEIR JOB
SELECT FIRST_NAME,SALARY,JOB_ID FROM EMPLOYEES WHERE (JOB_ID,SALARY)=ANY(SELECT JOB_ID,MIN(SALARY) FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY ALL OTHER THAN THOSE WHO GETS EITHER HIGHEST OR LOWER SALARY FOR THEIR MANAGER
SELECT FIRST_NAME, MANAGER_ID, SALARY FROM EMPLOYEES
WHERE (MANAGER_ID, SALARY) NOT IN ((SELECT MANAGER_ID,MIN(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID),(SELECT MANAGER_ID,MAX(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID));

15 DISPLAY THOSE EMPLOYEES WHO GETS HIGHER SALARY THAN THE AVERAGE SALARY PAID TO THOSE WHO WERE HIRED IN 98 OR 99. (DON'T USE LIKE)
SELECT FIRST_NAME,SALARY,TO_CHAR(HIRE_DATE,'RR') FROM EMPLOYEES WHERE SALARY > (SELECT AVG(SALARY) FROM EMPLOYEES WHERE TO_CHAR(HIRE_DATE,'RR') IN (98,99));

SELECT FIRST_NAME,DEPARTMENT_ID,SALARY FROM EMPLOYEES WHERE SALARY>(SELECT AVG(SALARY) FROM EMPLOYEES) AND DEPARTMENT_ID=80;

18 DISPLAY ALL IT_PROG WHO GETS LOWER SALARY THAN THE AVERAGE SALARY PAID BY COMPANY.
SELECT FIRST_NAME,JOB_ID,SALARY FROM EMPLOYEES WHERE SALARY<(SELECT AVG(SALARY) FROM EMPLOYEES) AND JOB_ID='IT_PROG';

19 DISPLAY EMPLOYEES WHO REPORTS TO 124 WHO GETS LOWER SALARY THAN THE AVERAGE SALARY PAID BY COMPANY.
SELECT FIRST_NAME,MANAGER_ID,SALARY FROM EMPLOYEES WHERE SALARY<(SELECT AVG(SALARY) FROM EMPLOYEES) AND MANAGER_ID=124;

20 DISPLAY ALL OTHER THAN THOSE WHO GETS HIGHEST SALARY FOR THEIR MANAGER.

WHERE (MANAGER_ID, SALARY) NOT IN (SELECT MANAGER_ID,MAX(SALARY) FROM EMPLOYEES GROUP BY MANAGER_ID);
