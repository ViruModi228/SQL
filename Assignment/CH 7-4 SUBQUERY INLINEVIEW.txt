1 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , HIGHEST SALARY PAID TO THE CORRESPONDING DEPARTMENT_ID
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,MAXSAL
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) MAXSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

2 DISPLAY LAST_NAME , JOB_ID , SALARY , LOWEST SALARY PAID TO THE CORRESPONDING JOB_ID
SELECT LAST_NAME,JOB_ID,SALARY,LOWSAL
FROM EMPLOYEES NATURAL JOIN (SELECT JOB_ID,MIN(SALARY) MINSAL FROM EMPLOYEES GROUP BY JOB_ID);

3 DISPLAY FIRST_NAME , MANAGER_ID , SALARY , AVERAGE SALARY PAID TO THE CORRESPONDING MANAGER_ID
SELECT FIRST_NAME,MANAGER_ID,SALARY,AVGSAL
FROM EMPLOYEES NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) AVGSAL FROM EMPLOYEES GROUP BY MANAGER_ID);

4 DISPLAY FIRST_NAME , HIRE YEAR , SALARY , HIRE YEAR WISE LOWEST SALARY , HIRE YEAR WISE HIGHEST SALARY
SELECT FIRST_NAME,TO_CHAR(HIRE_DATE,'RRRR'),SALARY,MAXSAL,MINSAL
FROM EMPLOYEES NATURAL JOIN
(SELECT TO_CHAR(HIRE_DATE,'RRRR'),MAX(SALARY) MAXSAL,MIN(SALARY) MINSAL FROM EMPLOYEES GROUP BY TO_CHAR(HIRE_DATE,'RRRR'));

5 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPARTMENT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY    // JODDDDD
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,JOB_ID,MAXSAL,MINSAL
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) MAXSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID,MIN(SALARY) MINSAL FROM EMPLOYEES GROUP BY JOB_ID);

6 DISPLAY LAST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , HIGHEST SALARY FOR THE CORRESPONDING DEPARTMENT_ID
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,MAXSAL,MINSAL
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) MAXSAL,MIN(SALARY) MINSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

7 DISPLAY FIRST_NAME , JOB_ID , SALARY , HIGHEST SALARY , DIFFERENCE (HIGHEST SALARY - SALARY) FOR THE CORRESPONDING JOB_ID
SELECT FIRST_NAME,SALARY,JOB_ID,MAXSAL,MAXSAL-SALARY
FROM EMPLOYEES NATURAL JOIN (SELECT JOB_ID,MAX(SALARY) MAXSAL FROM EMPLOYEES GROUP BY JOB_ID);

8 DISPLAY FIRST_NAME , LAST_NAME , DEPARTMENT_ID , JOB_ID , NO. OF EMPLOYEES WORKING FOR DEPARTMENT_ID , NO. OF EMPLOYEES WORKING FOR JOB_ID
SELECT FIRST_NAME,LAST_NAME,DEPARTMENT_ID,JOB_ID,DEPT_COUNT,JOB_COUNT
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,COUNT(*) DEPT_COUNT FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID,COUNT(*) JOB_COUNT FROM EMPLOYEES GROUP BY JOB_ID);

9 DISPLAY FIRST_NAME , JOB_ID , SALARY , TOTAL SALARY PAID TO JOB_ID , %OF SALARY EMPLOYEE GET FOR THE CORRESPONDING JOB_ID // DONE ONLY
SELCT FIRST_NAME,JOB_ID,SALARY,TOTAL_SAL,PERC_SAL
FROM EMPLOYEES 
NATURAL JOIN (SELECT JOB_ID,SUM(SALARY) TOTAL_SAL,(SALARY*100)/SUM(SALARY) PERC_SAL FROM EMPLOYEES GROUP BY JOB_ID);

10 DISPLAY FIRST_NAME , DEPARTMENT_ID , SALARY , LOWEST SALARY , DIFFERENCE IN % FOR THE CORRESPONDING DEPARTMENT_ID
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,MINSAL,100-((SALARY*100)/SUMSAL) DIFFPERC_SAL
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID,MIN(SALARY) MINSAL,SUM(SALARY) SUMSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID);

11 DISPLAY FIRST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , JOB_ID , JOB WISE LOWEST SALARY
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,JOB_ID,MAXSAL,MINSAL
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) MAXSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID,MIN(SALARY) MINSAL FROM EMPLOYEES GROUP BY JOB_ID);

12 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES
SELECT FIRST_NAME,DEPARTMENT_ID,SALARY,MANAGER_ID,DEPTAVGSAL,MGR_COUNT
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,AVG(SALARY) DEPTAVGSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT MANAGER_ID,COUNT(*) MGR_COUNT FROM EMPLOYEES GROUP BY MANAGER_ID);

13 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES
SELECT LAST_NAME,JOB_ID,SALARY,MANAGER_ID,MGRAVGSAL,JOB_COUNT
FROM EMPLOYEES NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) MGRAVGSAL FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN (SELECT JOB_ID,COUNT(*) JOB_COUNT FROM EMPLOYEES GROUP BY JOB_ID);

14 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY
SELECT LAST_NAME,SALARY,DEPARTMENT_ID,JOB_ID,MANAGER_ID,DEPTMINSAL,JOBAVGSAL,MGRMAXSAL
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID,MIN(SALARY) DEPTMINSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT JOB_ID,AVG(SALARY) JOBAVGSAL FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT MANAGER_ID,MAX(SALARY) MGRMAXSAL FROM EMPLOYEES GROUP BY MANAGER_ID);

15 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE HIGHEST SALARY , DEPARTMENT_NAME
SELECT LAST_NAME,DEPARTMENT_ID,SALARY,MAXSAL,DNAME
FROM EMPLOYEES NATURAL JOIN (SELECT DEPARTMENT_ID,MAX(SALARY) MAXSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT DEPARTMENT_ID,DEPARTMENT_NAME DNAME FROM DEPARTMENTS GROUP BY DEPARTMENT_ID,DEPARTMENT_NAME);

16 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE AVERAGE SALARY , MANAGER_ID , MANAGER WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY
SELECT LAST_NAME,SALARY,DEPARTMENT_ID,MANAGER_ID,DEPTAVGSAL,DNAME,LCITY,MGR_COUNT
FROM EMPLOYEES 
NATURAL JOIN (SELECT DEPARTMENT_ID,AVG(SALARY) DEPTAVGSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT LOCATION_ID,DEPARTMENT_ID,DEPARTMENT_NAME DNAME FROM DEPARTMENTS GROUP BY DEPARTMENT_ID,DEPARTMENT_NAME,LOCATION_ID)
NATURAL JOIN (SELECT LOCATION_ID,CITY LCITY FROM LOCATIONS GROUP BY LOCATION_ID,CITY);

17 DISPLAY LAST_NAME , SALARY , MANAGER_ID , MGR WISE AVERAGE SALARY , JOB_ID , JOB WISE NO. OF EMPLOYEES , DEPARTMENT_NAME , CITY , COUNTRY_NAME
SELECT LAST_NAME,SALARY,MANAGER_ID,MGRAVGSAL,JOB_ID,DNAME,CNAME,LCITY,JOB_COUNT
FROM EMPLOYEES 
NATURAL JOIN (SELECT MANAGER_ID,AVG(SALARY) MGRAVGSAL FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN (SELECT JOB_ID,COUNT(*) JOB_COUNT FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT LOCATION_ID,DEPARTMENT_ID,DEPARTMENT_NAME DNAME FROM DEPARTMENTS GROUP BY DEPARTMENT_ID,DEPARTMENT_NAME,LOCATION_ID)
NATURAL JOIN (SELECT COUNTRY_ID,LOCATION_ID,CITY LCITY FROM LOCATIONS GROUP BY LOCATION_ID,CITY,COUNTRY_ID)
NATURAL JOIN (SELECT COUNTRY_ID,COUNTRY_NAME CNAME FROM COUNTRIES GROUP BY COUNTRY_ID,COUNTRY_NAME);

18 DISPLAY LAST_NAME , SALARY , DEPARTMENT_ID , DEPT WISE LOWEST SALARY , JOB_ID , JOB WISE AVERAGE SALARY, MANAGER_ID , MANAGER WISE HIGHEST SALARY , DEPARTMENT_NAME , CITY , COUNTRY_NAME , REGION_NAME
SELECT LAST_NAME,SALARY,MANAGER_ID,MGRHIGHSAL,JOB_ID,JOBAVGSAL,DEPARTMENT_ID,DEPTMINSAL,DNAME,CNAME,LCITY,RNAME
FROM EMPLOYEES 
NATURAL JOIN (SELECT MANAGER_ID,MAX(SALARY) MGRHIGHSAL FROM EMPLOYEES GROUP BY MANAGER_ID)
NATURAL JOIN (SELECT JOB_ID,AVG(SALARY) JOBAVGSAL FROM EMPLOYEES GROUP BY JOB_ID)
NATURAL JOIN (SELECT DEPARTMENT_ID,MIN(SALARY) DEPTMINSAL FROM EMPLOYEES GROUP BY DEPARTMENT_ID)
NATURAL JOIN (SELECT LOCATION_ID,DEPARTMENT_ID,DEPARTMENT_NAME DNAME FROM DEPARTMENTS GROUP BY DEPARTMENT_ID,DEPARTMENT_NAME,LOCATION_ID)
NATURAL JOIN (SELECT COUNTRY_ID,LOCATION_ID,CITY LCITY FROM LOCATIONS GROUP BY LOCATION_ID,CITY,COUNTRY_ID)
NATURAL JOIN (SELECT REGION_ID,COUNTRY_ID,COUNTRY_NAME CNAME FROM COUNTRIES GROUP BY COUNTRY_ID,COUNTRY_NAME,REGION_ID)
NATURAL JOIN (SELECT REGION_ID,REGION_NAME RNAME FROM REGIONS GROUP BY REGION_ID,REGION_NAME);
